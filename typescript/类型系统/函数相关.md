### 定义函数类型

```typescript
// 1. 函数类型表达式
const fun: (a: string) => void = function (a) {
...
}

// 2. 调用签名，可以定义函数属性
type DescribableFunction = {
	description: string;
	(someArg: number): boolean;
};

// 3. 构造签名，定义需要new的函数
type SomeConstructor = {
	new (s: string): SomeObject;
};
```

### 函数重载
1. 指定一个函数的多种调用方式
2. 实现签名对外部是不可见的，所以需要在上方提供两个或多个重载签名给外部
```typescript
// 1. 定义重载函数
// 重载签名
function makeDate(timestamp: number): Date;
function makeDate(m: number, d: number, y: number): Date;
// 实现签名
function makeDate(mOrTimestamp: number, d?: number, y?: number): Date {
	if (d !== undefined && y !== undefined) {
		return new Date(y, mOrTimestamp, d);
	} else {
		return new Date(mOrTimestamp);
	}
}

// 2. 定义重载接口
interface MakeDate {
  (timestamp: number): Date;
  (m: number, d: number, y: number): Date;
}
const makeDate: MakeDate = (m: number, d?: number, y?: number) => {
  return new Date();
};

// 调用函数
makeDate(17584526352)
// 或者
makeDate(3, 10, 2025)
```

### 函数this类型
1. 规定函数体中this类型，同时规范了外部调用环境
```typescript
// 定义第一个参数名为this的类型
function (this: User) {
	return this.admin;
}

// ×××××××××箭头函数不能包含 "this" 参数××××××××××
const testFun = (this: string) => {
	...
}
```
